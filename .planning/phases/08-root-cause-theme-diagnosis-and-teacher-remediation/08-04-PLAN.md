```ts
import {
  pgTable,
  text,
  timestamp,
  jsonb,
  uniqueIndex,
  index,
} from "drizzle-orm/pg-core";
import { classes } from "./classes";
import type { LessonPlan } from "@mindmap/llm";  // mirror of Plan 08-03's Zod type

export const themeLessonPlans = pgTable(
  "theme_lesson_plans",
  {
    id: text("id")
      .primaryKey()
      .$defaultFn(() => crypto.randomUUID()),
    classId: text("class_id")
      .notNull()
      .references(() => classes.id, { onDelete: "cascade" }),
    themeId: text("theme_id").notNull(),                  // slug — NOT an FK
    dataHash: text("data_hash").notNull(),                // SHA-256 hex
    lessonPlan: jsonb("lesson_plan").$type<LessonPlan>().notNull(),
    generatedAt: timestamp("generated_at", { withTimezone: true }).defaultNow(),
    updatedAt: timestamp("updated_at", { withTimezone: true }).defaultNow(),
  },
  (t) => [
    uniqueIndex("theme_lesson_plans_class_theme_hash_idx").on(
      t.classId, t.themeId, t.dataHash
    ),
    index("theme_lesson_plans_class_theme_idx").on(t.classId, t.themeId),
  ]
);
```

dataHash + sha256Hex (08-RESEARCH.md §C7, exact code):
```ts
async function sha256Hex(input: string): Promise<string> {
  const bytes = new TextEncoder().encode(input);
  const digest = await crypto.subtle.digest("SHA-256", bytes);
  return Array.from(new Uint8Array(digest))
    .map((b) => b.toString(16).padStart(2, "0"))
    .join("");
}

export async function computeDataHash(
  tuples: Array<{
    misconceptionId: string;
    studentCount: number;
    unresolvedCount: number;
  }>
): Promise<string> {
  const sorted = [...tuples].sort((a, b) =>
    a.misconceptionId.localeCompare(b.misconceptionId)
  );
  const canonical = JSON.stringify(
    sorted.map((t) => [t.misconceptionId, t.studentCount, t.unresolvedCount])
  );
  return sha256Hex(canonical);
}
```

getOrGenerateLessonPlan (08-RESEARCH.md §C7):
```ts
export async function getOrGenerateLessonPlan(
  classId: string,
  themeId: string,
  opts: { forceRegenerate?: boolean } = {}
): Promise<LessonPlan> {
  // 1. Auth + ownership check (copy from dashboard.ts:70–76 pattern)
  const session = await auth();
  if (!session?.user?.id) throw new Error("Unauthorized");
  const cls = await db.query.classes.findFirst({
    where: and(eq(classes.id, classId), eq(classes.teacherId, session.user.id)),
  });
  if (!cls) throw new Error("Class not found or not owned");

  // 2. Compute current fingerprint from existing diagnostic sessions
  const tuples = await getThemeFingerprint(classId, themeId);
  const dataHash = await computeDataHash(tuples);

  // 3. Cache lookup unless forced
  if (!opts.forceRegenerate) {
    const cached = await db.query.themeLessonPlans.findFirst({
      where: and(
        eq(themeLessonPlans.classId, classId),
        eq(themeLessonPlans.themeId, themeId),
        eq(themeLessonPlans.dataHash, dataHash)
      ),
    });
    if (cached) return cached.lessonPlan;
  }

  // 4. Cache miss (or forced) — anonymized LLM call
  const theme = getThemeById(themeId);
  if (!theme) throw new Error("Theme not found");
  const constituents = getMisconceptionsByTheme(themeId);
  const fresh = await generateLessonPlan({
    theme,
    studentsAffected: tuples.reduce((sum, t) => sum + t.studentCount, 0),
    gradeBand: cls.gradeBand,  // adapt to actual cls field name
    constituentMisconceptions: constituents.map((c) => ({
      id: c.id,
      name: c.name,
      description: c.description,
      confrontation_scenarios: c.confrontation_scenarios,
    })),
  });

  // 5. Insert NEW row (D-18: not update — preserves history)
  await db.insert(themeLessonPlans).values({
    classId,
    themeId,
    dataHash,
    lessonPlan: fresh,
  });

  return fresh;
}
```
