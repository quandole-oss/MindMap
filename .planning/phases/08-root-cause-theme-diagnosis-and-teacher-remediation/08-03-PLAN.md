    #!/usr/bin/env bash
    # PRIV-01 audit ŌĆö fails if student identifiers leak into LLM prompt builders.
    # Source of truth: 08-RESEARCH.md ┬¦B5 + CONTEXT.md D-13
    set -euo pipefail

    REPO_ROOT="$(cd "$(dirname "$0")/.." && pwd)"
    cd "$REPO_ROOT"

    PROMPT_FILES=(
      "packages/llm/src/prompts/analyze-student-themes.ts"
      "packages/llm/src/prompts/generate-lesson-plan.ts"
    )

    ACTION_FILES=(
      "apps/web/actions/themes.ts"
    )

    PATTERN='\b(email|studentId|userId|enrollment)\b'

    fail=0

    echo "PRIV-01 audit: scanning prompt builders..."
    for f in "${PROMPT_FILES[@]}"; do
      if [[ ! -f "$f" ]]; then
        echo "  SKIP (file not found): $f"
        continue
      fi
      # Strip comment lines (// and /* ... */ block lines starting with *), then grep
      hits=$(sed -E '/^\s*\/\//d; /^\s*\*/d; /^\s*\/\*/d' "$f" | grep -nE "$PATTERN" || true)
      if [[ -n "$hits" ]]; then
        echo "  FAIL: $f"
        echo "$hits" | sed 's/^/    /'
        fail=1
      else
        echo "  OK:   $f"
      fi
    done

    echo ""
    echo "PRIV-01 audit: scanning server actions for student identifier leaks in return paths..."
    for f in "${ACTION_FILES[@]}"; do
      if [[ ! -f "$f" ]]; then
        echo "  SKIP (file not found): $f"
        continue
      fi
