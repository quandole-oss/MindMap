    2. Create `apps/web/actions/questions.ts` — server actions for question-related queries:
       ```typescript
       "use server";
       import { auth } from "@/lib/auth";
       import { db, schema } from "@mindmap/db";
       import { and, eq, gte, lt, desc, sql } from "drizzle-orm";

       export async function hasAskedToday(): Promise<boolean> {
         const session = await auth();
         if (!session?.user?.id) return false;
         const now = new Date();
         const startOfDay = new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth(), now.getUTCDate()));
         const startOfTomorrow = new Date(startOfDay.getTime() + 86_400_000);
         const existing = await db.query.questions.findFirst({
           where: and(
             eq(schema.questions.userId, session.user.id),
             gte(schema.questions.createdAt, startOfDay),
             lt(schema.questions.createdAt, startOfTomorrow),
           ),
         });
         return !!existing;
       }

       export async function getTodayQuestion() {
         const session = await auth();
         if (!session?.user?.id) return null;
         const now = new Date();
         const startOfDay = new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth(), now.getUTCDate()));
         const startOfTomorrow = new Date(startOfDay.getTime() + 86_400_000);
         return db.query.questions.findFirst({
           where: and(
             eq(schema.questions.userId, session.user.id),
             gte(schema.questions.createdAt, startOfDay),
             lt(schema.questions.createdAt, startOfTomorrow),
           ),
         });
       }

       export async function getStudentGradeLevel(): Promise<number> {
         const session = await auth();
         if (!session?.user?.id) return 6;
         const enrollment = await db.query.classEnrollments.findFirst({
           where: eq(schema.classEnrollments.studentId, session.user.id),
           orderBy: (t, { desc }) => [desc(t.enrolledAt)],
         });
         return enrollment?.gradeLevel ?? 6;
       }
       ```

    3. Create `apps/web/app/api/ask/route.ts` — the streaming POST endpoint:
       - Import `streamText` from `'ai'`, `anthropic` from `'@ai-sdk/anthropic'`
       - Import `auth` from `'@/lib/auth'`
       - Import `db, schema` from `'@mindmap/db'`
       - Import `buildEnrichSystemPrompt, extractConcepts` from `'@mindmap/llm'`
       - Import `routeQuestion` from `'@mindmap/router'`
       - Import `and, eq, gte, lt` from `'drizzle-orm'`
       - Export `const maxDuration = 60;`
       - POST handler:
         a. Auth check — `auth()`, return 401 if no session
         b. Parse request body: `{ question }` — validate question is non-empty string, max 500 chars
         c. One-per-day check using UTC date range query on questions table — return 429 with `{error: "One question per day"}` if already asked
         d. Get grade level from most recent classEnrollment (default 6 if unenrolled)
         e. Call `streamText({ model: anthropic('claude-sonnet-4-20250514'), system: buildEnrichSystemPrompt(gradeLevel), prompt: question, onFinish: async ({ text }) => { ... } })`
         f. In onFinish callback (wrapped in try/catch):
            - Insert question row: `db.insert(schema.questions).values({ userId, text: question, aiResponse: text }).returning()`
            - Call `extractConcepts(question, text)` to get concept array
            - For each concept: call `routeQuestion(concept.name, gradeLevel, concept.domain)`
            - Update question row with routingMode and routingMisconceptionId (use first concept's routing decision, or 'enrich' if no concepts)
            - Insert each concept into concepts table with `status: 'unprobed'`
            - Insert concept_questions join rows
            - If routing is 'diagnose': `console.log('[router] diagnose: ${concept.name} -> ${decision.misconceptionId}')` (per CONTEXT.md: log but serve enrich in Phase 2)
            - On error: `console.error('[onFinish] concept extraction failed:', err)` — silent failure, do not surface to user
         g. Return `result.toUIMessageStreamResponse()`

    IMPORTANT: The API route must NEVER include student name, email, or user ID in the prompt or system message. Only `question` (user text) and `gradeLevel` (integer) go to Claude. This satisfies PRIV-01.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client browser -> POST /api/ask | Untrusted user input (question text) crosses to server |
| Server -> Anthropic Claude API | Server sends question + grade level to external LLM |
| Anthropic Claude API -> Server | LLM response rendered as markdown in browser |
| Server -> Database | Question and concept data persisted |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-08 | Spoofing | apps/web/app/api/ask/route.ts | mitigate | `auth()` check at route entry; userId from session only, never from request body |
| T-02-09 | Tampering | apps/web/app/api/ask/route.ts | mitigate | Question text validated: non-empty string, max 500 chars. Zod validation before processing. |
| T-02-10 | Denial of Service | apps/web/app/api/ask/route.ts | mitigate | One-per-day DB check returns 429; unique index prevents race condition bypass; maxDuration=60 prevents infinite streaming |
| T-02-11 | Information Disclosure | apps/web/app/api/ask/route.ts | mitigate | Only question text and gradeLevel (integer) sent to Claude. No name/email/userId in prompt or system message. PRIV-01 enforced. |
| T-02-12 | Tampering | apps/web/components/questions/answer-display.tsx | mitigate | react-markdown renders to safe React elements (no dangerouslySetInnerHTML). Custom component override prevents img rendering. No raw HTML passthrough. |
</threat_model>
