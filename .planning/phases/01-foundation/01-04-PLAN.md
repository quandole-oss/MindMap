**1. `generateJoinCode()` — internal helper**

Per CONTEXT.md: 6-character alphanumeric codes. Exclude ambiguous chars (0, O, 1, I) per Research Pattern 6.

```typescript
function generateJoinCode(): string {
  const chars = "ABCDEFGHJKLMNPQRSTUVWXYZ23456789";
  return Array.from(crypto.getRandomValues(new Uint8Array(6)))
    .map((n) => chars[n % chars.length])
    .join("");
}

async function createUniqueJoinCode(): Promise<string> {
  let code: string;
  let attempts = 0;
  do {
    code = generateJoinCode();
    attempts++;
    if (attempts > 10) throw new Error("Failed to generate unique join code");
    const existing = await db.query.classes.findFirst({
      where: eq(schema.classes.joinCode, code),
    });
    if (!existing) return code;
  } while (true);
}
```

**2. `createClassAction`** (AUTH-05, AUTH-08)

Input: class name, grade level (integer 0-12 per CONTEXT.md). Validates teacher role. Creates class with unique join code. Grade level set by teacher (CONTEXT.md locked decision).

```typescript
const createClassSchema = z.object({
  name: z.string().min(1).max(100),
  gradeLevel: z.coerce.number().int().min(0).max(12), // 0=K, 1-12
});

export async function createClassAction(formData: FormData) {
  const session = await auth();
  if (!session?.user?.id) return { error: "Not authenticated" };

  // Verify teacher role
  const user = await db.query.users.findFirst({
    where: eq(schema.users.id, session.user.id),
  });
  if (!user || user.role !== "teacher") return { error: "Only teachers can create classes" };

  const parsed = createClassSchema.safeParse({
    name: formData.get("name"),
    gradeLevel: formData.get("gradeLevel"),
  });
  if (!parsed.success) return { error: parsed.error.issues[0].message };

  const joinCode = await createUniqueJoinCode();

  const [newClass] = await db.insert(schema.classes).values({
    name: parsed.data.name,
    joinCode,
    teacherId: session.user.id,
    gradeLevel: parsed.data.gradeLevel,
  }).returning();

  revalidatePath("/teacher");
  return { success: true, classId: newClass.id, joinCode };
}
```

**3. `joinClassAction`** (AUTH-06, AUTH-08)

Input: 6-character join code. Validates student role. Looks up class by join code. Creates enrollment with grade level inherited from class (AUTH-08 per CONTEXT.md: "students inherit class grade level").

Error messages per UI-SPEC:
- Code not found: "That code didn't match any class. Double-check the code with your teacher."
- Already enrolled: "You're already a member of this class."

```typescript
const joinClassSchema = z.object({
  joinCode: z.string().length(6).toUpperCase(),
});

export async function joinClassAction(formData: FormData) {
  const session = await auth();
  if (!session?.user?.id) return { error: "Not authenticated" };

  const user = await db.query.users.findFirst({
    where: eq(schema.users.id, session.user.id),
  });
  if (!user || user.role !== "student") return { error: "Only students can join classes" };

  const parsed = joinClassSchema.safeParse({
    joinCode: (formData.get("joinCode") as string || "").toUpperCase(),
  });
  if (!parsed.success) return { error: parsed.error.issues[0].message };

  const targetClass = await db.query.classes.findFirst({
    where: eq(schema.classes.joinCode, parsed.data.joinCode),
  });
  if (!targetClass) {
    return { error: "That code didn't match any class. Double-check the code with your teacher." };
  }

  // Check if already enrolled
  const existingEnrollment = await db.query.classEnrollments.findFirst({
    where: and(
      eq(schema.classEnrollments.classId, targetClass.id),
      eq(schema.classEnrollments.studentId, session.user.id),
    ),
  });
  if (existingEnrollment) {
    return { error: "You're already a member of this class." };
  }

  // Enroll student with grade level inherited from class (AUTH-08)
  await db.insert(schema.classEnrollments).values({
    classId: targetClass.id,
    studentId: session.user.id,
    gradeLevel: targetClass.gradeLevel, // inherited from class
  });

  revalidatePath("/student");
  return { success: true, className: targetClass.name };
}
```

**4. `removeStudentAction`** (AUTH-07)

Input: enrollment ID. Validates teacher role. Verifies teacher owns the class. Deletes enrollment.

```typescript
export async function removeStudentAction(enrollmentId: string) {
  const session = await auth();
  if (!session?.user?.id) return { error: "Not authenticated" };

  const enrollment = await db.query.classEnrollments.findFirst({
    where: eq(schema.classEnrollments.id, enrollmentId),
    with: { class: true }, // need this relation - see note below
  });

  if (!enrollment) return { error: "Enrollment not found" };

  // Verify teacher owns the class
  const targetClass = await db.query.classes.findFirst({
    where: and(
      eq(schema.classes.id, enrollment.classId),
      eq(schema.classes.teacherId, session.user.id),
    ),
  });
  if (!targetClass) return { error: "Not authorized" };

  await db.delete(schema.classEnrollments)
    .where(eq(schema.classEnrollments.id, enrollmentId));

  revalidatePath(`/teacher/classes/${enrollment.classId}/roster`);
  return { success: true };
}
```

Note: If Drizzle `with` relation requires explicit relation definitions, add relations in the schema. Alternatively, use a two-query approach (fetch enrollment, then fetch class) to avoid needing relation config.

export async function getClassRoster(classId: string) {
  const session = await auth();
  if (!session?.user?.id) return { error: "Not authenticated" };

  const targetClass = await db.query.classes.findFirst({
    where: and(
      eq(schema.classes.id, classId),
      eq(schema.classes.teacherId, session.user.id),
    ),
  });
  if (!targetClass) return { error: "Not authorized" };

  const enrollments = await db
    .select({
      enrollmentId: schema.classEnrollments.id,
      studentId: schema.classEnrollments.studentId,
      gradeLevel: schema.classEnrollments.gradeLevel,
      enrolledAt: schema.classEnrollments.enrolledAt,
      studentName: schema.users.name,
      studentEmail: schema.users.email,
    })
    .from(schema.classEnrollments)
    .innerJoin(schema.users, eq(schema.classEnrollments.studentId, schema.users.id))
    .where(eq(schema.classEnrollments.classId, classId));

  return {
    class: targetClass,
    students: enrollments,
  };
}

export async function getTeacherClasses() {
  const session = await auth();
  if (!session?.user?.id) return [];

  return db.query.classes.findMany({
    where: eq(schema.classes.teacherId, session.user.id),
  });
}

export async function getStudentEnrollments() {
  const session = await auth();
  if (!session?.user?.id) return [];

  return db
    .select({
      enrollmentId: schema.classEnrollments.id,
      classId: schema.classes.id,
      className: schema.classes.name,
      gradeLevel: schema.classEnrollments.gradeLevel,
      joinCode: schema.classes.joinCode,
    })
    .from(schema.classEnrollments)
    .innerJoin(schema.classes, eq(schema.classEnrollments.classId, schema.classes.id))
    .where(eq(schema.classEnrollments.studentId, session.user.id));
}

| Boundary | Description |
|----------|-------------|
| Client form -> Server Action | Join code, class name, grade level cross trust boundary |
| Teacher role check | Server actions verify role before class operations |
| Class ownership check | Roster/remove actions verify teacher owns the class |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Spoofing | Join code brute force | accept | 30^6 = 729M possible codes; 10-attempt limit in generator; no rate limit needed at capstone scale |
| T-04-02 | Elevation of Privilege | Student accessing createClassAction | mitigate | Server action checks `user.role !== "teacher"` and returns error before any DB write |
| T-04-03 | Elevation of Privilege | Teacher modifying another teacher's class | mitigate | `getClassRoster` and `removeStudentAction` verify `teacherId === session.user.id` |
| T-04-04 | Information Disclosure | Student viewing other class rosters | mitigate | Roster page fetches via `getClassRoster` which checks teacher ownership; students cannot access roster endpoint |
| T-04-05 | Tampering | Grade level manipulation | mitigate | Zod validates gradeLevel as integer 0-12 in createClassAction; enrollment inherits from class, not from student input |
</threat_model>
