```typescript
// Collect resolved concept IDs for edge creation
const resolvedConceptIds: string[] = [];

for (const { concept, decision } of routingDecisions) {
  if (decision.mode === "diagnose") {
    console.log(`[router] diagnose: ${concept.name} -> ${decision.misconceptionId}`);
  }

  let conceptId: string;

  try {
    // Stage 1: Generate embedding for this concept
    const embedding = await generateEmbedding(concept.name);

    // Stage 1b: Find similar existing concepts via pgvector ANN
    const candidates = await findSimilarConcepts(userId, embedding, 0.85, 5);

    // Check for auto-merge (distance < 0.08 = similarity > 0.92)
    const autoMerge = candidates.find(c => (c.distance as number) < 0.08);
    if (autoMerge) {
      // Auto-merge: increment visitCount on existing concept
      await db.update(schema.concepts)
        .set({ visitCount: sql`${schema.concepts.visitCount} + 1` })
        .where(eq(schema.concepts.id, autoMerge.id));
      conceptId = autoMerge.id;
      console.log(`[dedup] auto-merge: "${concept.name}" -> "${autoMerge.name}" (distance: ${autoMerge.distance})`);
    } else {
      // Check for borderline (0.08 <= distance < 0.15 = 0.85-0.92 similarity)
      const borderline = candidates.filter(c => (c.distance as number) >= 0.08 && (c.distance as number) < 0.15);
      if (borderline.length > 0) {
        // Stage 2: LLM disambiguation
        const result = await disambiguateConcept(concept.name, concept.domain, borderline as any, question);
        if (result.match && borderline[result.matchIndex]) {
          // LLM confirmed match — merge
          const matchedId = borderline[result.matchIndex].id;
          await db.update(schema.concepts)
            .set({ visitCount: sql`${schema.concepts.visitCount} + 1` })
            .where(eq(schema.concepts.id, matchedId));
          conceptId = matchedId;
          console.log(`[dedup] LLM-merge: "${concept.name}" -> "${borderline[result.matchIndex].name}"`);
        } else {
          // LLM says different concept — create new
          const [saved] = await db.insert(schema.concepts)
            .values({ userId, name: concept.name, domain: concept.domain, status: "unprobed", embedding, visitCount: 1 })
            .returning();
          conceptId = saved.id;
          console.log(`[dedup] LLM-split: "${concept.name}" is new`);
        }
      } else {
        // No similar concepts — create new
        const [saved] = await db.insert(schema.concepts)
          .values({ userId, name: concept.name, domain: concept.domain, status: "unprobed", embedding, visitCount: 1 })
          .returning();
        conceptId = saved.id;
        console.log(`[dedup] new: "${concept.name}"`);
      }
    }
  } catch (embedErr) {
    // Embedding or dedup failed — fall back to inserting without dedup (silent to student per UI-SPEC)
    console.error(`[dedup] embedding/dedup failed for "${concept.name}":`, embedErr);
    const [saved] = await db.insert(schema.concepts)
      .values({ userId, name: concept.name, domain: concept.domain, status: "unprobed", visitCount: 1 })
      .returning();
    conceptId = saved.id;
  }

  // Always create the concept-question join
  await db.insert(schema.conceptQuestions).values({
    conceptId,
    questionId: savedQuestion.id,
  });

  resolvedConceptIds.push(conceptId);
}

// Create curiosity_link edges between all concepts from this question
if (resolvedConceptIds.length >= 2) {
  try {
    await createConceptEdges(resolvedConceptIds, "curiosity_link");
  } catch (edgeErr) {
    console.error("[dedup] edge creation failed:", edgeErr);
  }
}
```
