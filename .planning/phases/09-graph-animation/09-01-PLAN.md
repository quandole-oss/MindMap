  <action>
Modify `apps/web/components/graph/use-graph-layout.ts` to support incremental layout that preserves existing node positions.

**Changes:**

1. Add a `useRef` to store the previous layout result (positions by node ID).

2. Inside the `useMemo`, detect whether this is an initial layout (no previous positions) or an incremental layout (previous positions exist).

3. **Initial layout** (no change from current behavior): Run `simulation.tick(300)` from scratch.

4. **Incremental layout** (new nodes added):
   a. Before running the simulation, copy `x`, `y`, `z` from the previous layout result onto matching nodes (by ID).
   b. Set `fx`, `fy`, `fz` on existing nodes (pin them) — these tell d3-force to keep them fixed.
   c. For new nodes (no previous position), set initial position near the centroid of connected existing nodes (if they have edges to existing nodes) or at a random offset from the graph center.
   d. Run `simulation.tick(50)` — new nodes settle into position while existing nodes stay pinned.
   e. Remove `fx`, `fy`, `fz` from all nodes (unpin).
   f. Run `simulation.tick(50)` more — everything relaxes slightly.

5. After the simulation runs, store the result in the ref for the next incremental run.

6. The hook signature stays the same: `useGraphLayout(nodes, edges)` returns `LayoutNode[]`.

Implementation:

```typescript
"use client";

import { useMemo, useRef } from "react";
import {
  forceSimulation,
  forceLink,
  forceManyBody,
  forceCenter,
  forceCollide,
  forceX,
  forceY,
  forceZ,
} from "d3-force-3d";
import type { GraphNode, GraphEdge } from "@/actions/graph";

export interface LayoutNode extends GraphNode {
  x: number;
  y: number;
  z: number;
}

function computeDomainPositions(
  domains: string[]
): Map<string, [number, number, number]> {
  const sorted = [...new Set(domains)].sort();
  const map = new Map<string, [number, number, number]>();
  const n = sorted.length;

  if (n === 0) return map;
  if (n === 1) {
    map.set(sorted[0], [0, 0, 0]);
    return map;
  }

  const R = 120;
  const goldenRatio = (1 + Math.sqrt(5)) / 2;

  for (let i = 0; i < n; i++) {
    const theta = Math.acos(1 - (2 * (i + 0.5)) / n);
    const phi = (2 * Math.PI * i) / goldenRatio;
    map.set(sorted[i], [
      R * Math.sin(theta) * Math.cos(phi),
      R * Math.sin(theta) * Math.sin(phi),
      R * Math.cos(theta),
    ]);
  }

  return map;
}

export function useGraphLayout(
  nodes: GraphNode[],
  edges: GraphEdge[]
): LayoutNode[] {
  // Store previous layout positions for incremental updates (D-17, D-18)
  const prevPositionsRef = useRef<Map<string, { x: number; y: number; z: number }>>(
    new Map()
  );

  return useMemo(() => {
    if (nodes.length === 0) return [];

    const prevPositions = prevPositionsRef.current;
    const isIncremental = prevPositions.size > 0;

    // Deep-clone nodes and edges — d3-force-3d mutates objects in-place
    const simNodes = nodes.map((n) => ({
      ...n,
    })) as Array<GraphNode & { x: number; y: number; z: number; fx?: number | null; fy?: number | null; fz?: number | null }>;
    const simLinks = edges.map((e) => ({ ...e }));

    // Compute deterministic domain positions on a sphere
    const domainPositions = computeDomainPositions(nodes.map((n) => n.domain));

    if (isIncremental) {
      // D-17: Preserve existing node positions, pin them
      // Compute centroid of existing nodes for new node initial placement
      let cx = 0, cy = 0, cz = 0, existingCount = 0;

      for (const node of simNodes) {
        const prev = prevPositions.get(node.id);
        if (prev) {
          // Existing node: restore position and pin
          node.x = prev.x;
          node.y = prev.y;
          node.z = prev.z;
          node.fx = prev.x;
          node.fy = prev.y;
          node.fz = prev.z;
          cx += prev.x;
          cy += prev.y;
          cz += prev.z;
          existingCount++;
        }
      }

      if (existingCount > 0) {
        cx /= existingCount;
        cy /= existingCount;
        cz /= existingCount;
      }

      // Build adjacency for finding connected existing nodes
      const nodeIdxMap = new Map(simNodes.map((n, i) => [n.id, i]));

      for (const node of simNodes) {
        if (prevPositions.has(node.id)) continue;

        // New node: place near connected existing nodes or at centroid + random offset
        let nearX = cx, nearY = cy, nearZ = cz;
        let foundNeighbor = false;

        for (const edge of edges) {
          const neighborId = edge.source === node.id ? edge.target : edge.target === node.id ? edge.source : null;
          if (neighborId) {
            const neighborPrev = prevPositions.get(neighborId);
            if (neighborPrev) {
              nearX = neighborPrev.x;
              nearY = neighborPrev.y;
              nearZ = neighborPrev.z;
              foundNeighbor = true;
              break;
            }
          }
        }

        // Add random jitter so new nodes don't overlap
        const jitter = foundNeighbor ? 30 : 60;
        node.x = nearX + (Math.random() - 0.5) * jitter;
        node.y = nearY + (Math.random() - 0.5) * jitter;
        node.z = nearZ + (Math.random() - 0.5) * jitter;
      }
    }

    const simulation = forceSimulation(simNodes, 3)
      .force(
        "link",
        forceLink(simLinks)
          .id((d: any) => d.id)
          .distance((link: any) =>
            link.edgeType === "bridge" ? 100 : 60
          )
          .strength((link: any) => 0.3 + (link.weight ?? 0.5) * 0.7)
      )
      .force("charge", forceManyBody().strength(-180))
      .force("center", forceCenter(0, 0, 0))
      .force("collide", forceCollide().radius(16))
      .force(
        "domainX",
        forceX((d: any) => domainPositions.get(d.domain)?.[0] ?? 0).strength(0.15)
      )
      .force(
        "domainY",
        forceY((d: any) => domainPositions.get(d.domain)?.[1] ?? 0).strength(0.15)
      )
      .force(
        "domainZ",
        forceZ((d: any) => domainPositions.get(d.domain)?.[2] ?? 0).strength(0.15)
      )
      .stop();

    if (isIncremental) {
      // D-18: Incremental — 50 ticks with pins, then 50 without
      simulation.tick(50);

      // Release all pins
      for (const node of simNodes) {
        node.fx = null;
        node.fy = null;
        node.fz = null;
      }

      simulation.tick(50);
    } else {
      // Initial layout — full 300 ticks
      simulation.tick(300);
    }

    // Store positions for next incremental update
    const newPositions = new Map<string, { x: number; y: number; z: number }>();
    for (const node of simNodes) {
      newPositions.set(node.id, { x: node.x, y: node.y, z: node.z });
    }
    prevPositionsRef.current = newPositions;

    return simNodes as LayoutNode[];
  }, [nodes, edges]);
}
```

Key implementation notes:
- `prevPositionsRef` persists across renders without causing re-renders
- `isIncremental` is `true` after the first layout run, so subsequent data changes preserve positions
- New nodes are placed near their connected existing neighbors for natural positioning
- The 50+50 tick strategy (D-18) lets new nodes find their spot without disrupting the existing graph
- All existing force parameters (charge, link distance, domain galaxies) are preserved

Modify `apps/web/components/graph/solar-nodes.tsx` to over-allocate the InstancedMesh buffer and remove the `key={layoutNodes.length}` pattern that forces mesh remount.

**Changes:**

1. **Define a growth buffer constant:**
```typescript
const GROWTH_BUFFER = 20;
```

2. **Replace `key={layoutNodes.length}`** on the `<instancedMesh>` element. Remove the `key` prop entirely. The mesh should persist across node count changes.

3. **Change `args` from `layoutNodes.length` to `layoutNodes.length + GROWTH_BUFFER`:**
```typescript
args={[undefined, undefined, layoutNodes.length + GROWTH_BUFFER]}
```

BUT since we are removing the key-based remount, we need to handle buffer exhaustion. Track the allocated buffer size in a ref and only force a remount when the buffer is actually exhausted.

4. **Set `mesh.count` to `layoutNodes.length`** in the useEffect where matrices are initialized:
```typescript
meshRef.current.count = layoutNodes.length;
```

5. **Handle the case where layoutNodes.length exceeds the allocated buffer:** Use a ref to track the current buffer capacity. Use a key based on the buffer capacity (not node count) so remounts only happen when the buffer is exhausted (rare).

Implementation for the component:

```typescript
"use client";

import { useRef, useEffect, useMemo, useCallback, useState } from "react";

import * as THREE from "three";
import type { LayoutNode } from "./use-graph-layout";
import type { GraphNode } from "@/actions/graph";

const NODE_COLORS: Record<string, THREE.Color> = {
  healthy: new THREE.Color("#0d9488"),
  misconception: new THREE.Color("#dc2626"),
  unprobed: new THREE.Color("#71717a"),
  bridge: new THREE.Color("#7c3aed"),
};

function getNodeColor(node: GraphNode): THREE.Color {
  if (node.isBridge) return NODE_COLORS.bridge.clone().multiplyScalar(6.0);
  if (node.status === "misconception") return NODE_COLORS.misconception.clone().multiplyScalar(5.0);
  if (node.status === "healthy") return NODE_COLORS.healthy.clone().multiplyScalar(4.0);
  return NODE_COLORS.unprobed.clone().multiplyScalar(3.0);
}

export function getNodeRadius(node: GraphNode): number {
  const MIN = 1.5, MAX = 14;
  const t = node.importance ?? 0;
  return MIN + t * t * (MAX - MIN);
}

/** D-15: Over-allocate by this many instances beyond current count */
const GROWTH_BUFFER = 20;

interface SolarNodesProps {
  layoutNodes: LayoutNode[];
  onNodeClick: (nodeId: string) => void;
  onNodeHover: (node: LayoutNode | null) => void;
  highlightNodeId?: string | null;
  meshRef?: React.MutableRefObject<THREE.InstancedMesh | null>;
  /** Set of node IDs that are newly added and should start at scale 0 */
  newNodeIds?: Set<string>;
}

export function SolarNodes({
  layoutNodes,
  onNodeClick,
  onNodeHover,
  highlightNodeId,
  meshRef: externalMeshRef,
  newNodeIds,
}: SolarNodesProps) {
  const internalMeshRef = useRef<THREE.InstancedMesh>(null!);
  const meshRef = internalMeshRef;

  const dummy = useMemo(() => new THREE.Object3D(), []);

  const [meshReady, setMeshReady] = useState(false);

  // D-15: Track buffer capacity. Only remount when buffer is exhausted.
  const bufferCapacity = useRef(0);

  // Compute required capacity
  const requiredCapacity = layoutNodes.length + GROWTH_BUFFER;

  // Only update the buffer key when capacity is actually exceeded
  // This prevents unnecessary remounts when nodes are added within the buffer
  if (bufferCapacity.current === 0 || layoutNodes.length > bufferCapacity.current) {
    bufferCapacity.current = requiredCapacity;
  }

  // Initial matrix + color setup whenever layout or mesh changes
  useEffect(() => {
    if (!meshRef.current || layoutNodes.length === 0) return;

    // D-15: Set visible count to actual node count (not buffer size)
    meshRef.current.count = layoutNodes.length;

    layoutNodes.forEach((node, i) => {
      // If this is a new node, start at scale 0 (animation will scale it up)
      const isNew = newNodeIds?.has(node.id) ?? false;
      const scale = isNew ? 0 : getNodeRadius(node);

      dummy.position.set(node.x, node.y, node.z);
      dummy.scale.setScalar(scale);
      dummy.updateMatrix();
      meshRef.current.setMatrixAt(i, dummy.matrix);
      meshRef.current.setColorAt(i, getNodeColor(node));
    });

    meshRef.current.instanceMatrix.needsUpdate = true;
    if (meshRef.current.instanceColor) {
      meshRef.current.instanceColor.needsUpdate = true;
    }
    meshRef.current.computeBoundingBox();
    meshRef.current.computeBoundingSphere();
  }, [layoutNodes, dummy, meshReady, newNodeIds]);

  const setMeshRef = useCallback(
    (instance: THREE.InstancedMesh | null) => {
      internalMeshRef.current = instance!;
      if (externalMeshRef) {
        externalMeshRef.current = instance;
      }
      if (instance) setMeshReady(true);
    },
    [externalMeshRef]
  );

  if (layoutNodes.length === 0) return null;

  return (
    <instancedMesh
      key={bufferCapacity.current}
      ref={setMeshRef}
      args={[undefined, undefined, bufferCapacity.current]}
      onClick={(e) => {
        e.stopPropagation();
        const node = layoutNodes[e.instanceId!];
        if (node) onNodeClick(node.id);
      }}
      onPointerOver={(e) => {
        e.stopPropagation();
        onNodeHover(layoutNodes[e.instanceId!] ?? null);
      }}
      onPointerOut={() => onNodeHover(null)}
    >
      <sphereGeometry args={[1, 16, 16]} />
      <meshBasicMaterial
        toneMapped={false}
      />
    </instancedMesh>
  );
}
```

Key changes from current code:
- `GROWTH_BUFFER = 20` constant (D-15)
- `bufferCapacity` ref tracks allocated size; `key` is now `bufferCapacity.current` — only changes when buffer is exhausted (rare), not on every node addition
- `mesh.count = layoutNodes.length` in useEffect (D-15) — only visible nodes are rendered
- `newNodeIds` prop added — when provided, new nodes start at scale 0 (invisible) so SolarScene can animate them in (D-16 prep)
- Removed the old `key={layoutNodes.length}` pattern that caused full remount on every addition
