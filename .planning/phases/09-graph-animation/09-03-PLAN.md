**Animation initialization (in useEffect, triggered when newNodeIds changes):**

When `animateEntry` is true and `newNodeIds` has entries, set up the animation:

```typescript
useEffect(() => {
  if (!animateEntry || !newNodeIds || newNodeIds.size === 0) return;
  if (layoutNodes.length === 0) return;

  isAnimating.current = true;
  animationStartTime.current = -1; // Will be set on first useFrame

  // Find instance indices for new nodes and set up staggered birth times
  // (actual clock times will be computed on first frame)
  const newIndices: number[] = [];
  layoutNodes.forEach((node, idx) => {
    if (newNodeIds.has(node.id)) {
      newIndices.push(idx);
    }
  });

  // Store indices for birth timestamp setup in the first useFrame call
  birthTimestamps.current.clear();
  newIndices.forEach((idx, i) => {
    // -1 means "compute on first frame using clock.elapsedTime + stagger"
    birthTimestamps.current.set(idx, -(i + 1)); // negative = "pending, stagger index i"
  });

  // Camera: compute bounding sphere of new nodes + nearby existing nodes
  const newNodes = layoutNodes.filter(n => newNodeIds.has(n.id));
  if (newNodes.length > 0) {
    let cx = 0, cy = 0, cz = 0;
    for (const n of newNodes) { cx += n.x; cy += n.y; cz += n.z; }
    cx /= newNodes.length; cy /= newNodes.length; cz /= newNodes.length;

    let maxDist = 0;
    for (const n of newNodes) {
      const d = Math.sqrt((n.x - cx) ** 2 + (n.y - cy) ** 2 + (n.z - cz) ** 2);
      if (d > maxDist) maxDist = d;
    }

    // D-09: Camera pull back to frame new content with margin
    const fov = (camera as THREE.PerspectiveCamera).fov ?? 60;
    const fovRad = (fov * Math.PI) / 180;
    const dist = Math.max((maxDist + 60) / Math.tan(fovRad / 2), 80);

    targetPosition.current.set(cx + dist * 0.3, cy + dist * 0.2, cz + dist);
    targetLookAt.current.set(cx, cy, cz);
    isFlying.current = true;
  }
}, [animateEntry, newNodeIds, layoutNodes, camera]);
```

```typescript
// === Birth animation loop (D-07, D-16) ===
if (isAnimating.current && meshRef.current && birthTimestamps.current.size > 0) {
  const now = clock.elapsedTime;

  // Initialize birth times on first frame
  if (animationStartTime.current === -1) {
    animationStartTime.current = now;
    const entries = Array.from(birthTimestamps.current.entries());
    birthTimestamps.current.clear();
    sparkleNodes.current = [];

    for (const [idx, staggerMarker] of entries) {
      // D-07: 300ms stagger between nodes
      const staggerIndex = Math.abs(staggerMarker) - 1;
      const birthTime = now + staggerIndex * 0.3;
      birthTimestamps.current.set(idx, birthTime);

      // Set up sparkle data for this node
      const node = layoutNodes[idx];
      if (node) {
        sparkleNodes.current.push({
          x: node.x, y: node.y, z: node.z,
          startTime: birthTime,
          endTime: birthTime + 0.5, // D-10: 500ms duration
        });
      }
    }
  }

  let allComplete = true;

  for (const [idx, birthTime] of birthTimestamps.current) {
    const node = layoutNodes[idx];
    if (!node) continue;

    const elapsed = now - birthTime;
    if (elapsed < 0) {
      // Not yet born — keep at scale 0
      allComplete = false;
      continue;
    }

    // D-07: 500ms scale-up with ease-out cubic
    const DURATION = 0.5;
    const t = Math.min(elapsed / DURATION, 1);
    // Ease-out cubic: t = 1 - (1-t)^3 (UI-SPEC)
    const eased = 1 - Math.pow(1 - t, 3);

    const targetRadius = getNodeRadius(node);
    const currentScale = targetRadius * eased;

    pulseDummy.current.position.set(node.x, node.y, node.z);
    pulseDummy.current.scale.setScalar(currentScale);
    pulseDummy.current.updateMatrix();
    meshRef.current.setMatrixAt(idx, pulseDummy.current.matrix);

    if (t < 1) allComplete = false;
  }

  meshRef.current.instanceMatrix.needsUpdate = true;

  // Check if edge animation should start
  // D-08: After all nodes settled (~1s after last node starts)
  if (allComplete && edgeAnimationStart.current === -1) {
    edgeAnimationStart.current = now;
  }

  // Check if entire animation is complete (nodes done + edges done)
  if (allComplete && edgeAnimationStart.current > 0) {
    const edgeElapsed = now - edgeAnimationStart.current;
    // Edges need ~0.4s per edge, but they can overlap; allow 1s total
    if (edgeElapsed > 1.0) {
      isAnimating.current = false;
      birthTimestamps.current.clear();
      sparkleNodes.current = [];
      edgeAnimationStart.current = -1;
      // Signal animation complete (for growth summary overlay in 09-04)
      onAnimationComplete?.();
    }
  }
}

**Edge draw-in implementation:**

import { useRef } from "react";
```

```typescript
export function SolarEdges({ layoutNodes, edges, newEdgeKeys, edgeAnimationStartTime }: SolarEdgesProps) {
  // ... existing nodeIndex and sortedEdges logic

  // Ref to track animation progress per edge
  const edgeProgressRef = useRef<Map<string, number>>(new Map());

  // Track draw-in state via ref for useFrame
  const drawInActive = useRef(false);

  useFrame(({ clock }) => {
    if (!newEdgeKeys || newEdgeKeys.size === 0 || edgeAnimationStartTime === undefined || edgeAnimationStartTime < 0) {
      drawInActive.current = false;
      return;
    }

    const now = clock.elapsedTime;
    if (now < edgeAnimationStartTime) return;

    drawInActive.current = true;
    const elapsed = now - edgeAnimationStartTime;

    // Calculate progress for each new edge (400ms each, linear - D-08)
    for (const key of newEdgeKeys) {
      const progress = Math.min(elapsed / 0.4, 1); // D-08: 400ms per edge
      edgeProgressRef.current.set(key, progress);
    }
  });

  return (
    <>
      {sortedEdges.map((edge, i) => {
        const src = nodeIndex[edge.source];
        const tgt = nodeIndex[edge.target];
        if (!src || !tgt) return null;

        const { color, opacity, lineWidth } = getEdgeStyle(edge, src.domain, tgt.domain);

        // Determine if this edge is animating
        const edgeKey = edge.source < edge.target
          ? `${edge.source}:${edge.target}`
          : `${edge.target}:${edge.source}`;
        const isNewEdge = newEdgeKeys?.has(edgeKey) ?? false;

        if (isNewEdge && edgeAnimationStartTime !== undefined && edgeAnimationStartTime >= 0) {
          const progress = edgeProgressRef.current.get(edgeKey) ?? 0;
          if (progress <= 0) return null; // Not yet started

          // Draw-in: interpolate end point from source toward target
          const endX = src.x + (tgt.x - src.x) * progress;
          const endY = src.y + (tgt.y - src.y) * progress;
          const endZ = src.z + (tgt.z - src.z) * progress;

          return (
            <Line
              key={i}
              points={[
                [src.x, src.y, src.z],
                [endX, endY, endZ],
              ]}
              color={color}
              lineWidth={lineWidth}
              transparent
              opacity={opacity * progress}
            />
          );
        }

        // Existing edge — render normally
        return (
          <Line
            key={i}
            points={[
              [src.x, src.y, src.z],
              [tgt.x, tgt.y, tgt.z],
            ]}
            color={color}
            lineWidth={lineWidth}
            transparent
            opacity={opacity}
          />
        );
      })}
    </>
  );
}
```
